# Complete Calc Proof Chain for $$
E[\tau] = e
$$

```lean
import Mathlib.Topology.Algebra.InfiniteSum.Basic  -- HasSum, Summable, tsum_subtype, tsum_add_tsum_compl
import Mathlib.Data.Real.Basic                    -- ℝ, exp
import Mathlib.Tactic.Calc                         -- calc mode
import Mathlib.Tactic.Ring                         -- ring
import Mathlib.Tactic.Simp                         -- simp
import Mathlib.Tactic.Omega                        -- omega for n-2 arithmetic

open Filter

/-- Step 1: Unfold the definition of `expected_hitting_time`. -/
lemma expected_value_as_series :
  expected_hitting_time = ∑' n : ℕ, n * hitting_time_pmf n := by
  -- By definition of expectation for discrete distribution
  unfold expected_hitting_time
  rfl

/-- Step 2: Substitute the PMF definition. -/
lemma expand_pmf_in_sum :
  ∑' n : ℕ, n * hitting_time_pmf n =
    ∑' n : ℕ, if n ≤ 1 then 0 else n * ((n - 1 : ℝ) / n.factorial) := by
  congr 1
  funext n
  simp only [hitting_time_pmf]
  split_ifs with h
  · simp
  · rfl

/-- Step 3: Eliminate the zero terms \(n = 0,1\). -/
lemma eliminate_zero_terms :
  ∑' n : ℕ, if n ≤ 1 then 0 else n * ((n - 1 : ℝ) / n.factorial) =
    ∑' n : {n // 2 ≤ n}, n * ((n - 1 : ℝ) / n.factorial) := by
  -- Restrict to `n ≥ 2`, since the terms for `n = 0,1` are zero
  simp only [tsum_subtype]
  · rfl
  · intro n hn
    have : n ≤ 1 := (Nat.not_le.1 hn).le
    simp [this]

/-- Step 4: Simplify each term \(n\cdot\frac{n-1}{n!} = \frac1{(n-2)!}\). -/
lemma simplify_factorial_expression (n : ℕ) (hn : 2 ≤ n) :
  (n : ℝ) * ((n - 1 : ℝ) / n.factorial) = 1 / (n - 2).factorial := by
  -- algebraic manipulation
  calc
    (n : ℝ) * ((n - 1 : ℝ) / n.factorial)
      = (n : ℝ) * (n - 1 : ℝ) / n.factorial := by ring
    _ = (n * (n - 1) : ℝ) / (n * (n - 1).factorial) := by
      -- `n! = n * (n-1)!`
      have : n.factorial = n * (n - 1).factorial := by
        rw [Nat.factorial_succ, Nat.succ_eq_add_one]; rfl
      simp [this]
    _ = (n - 1 : ℝ) / (n - 1).factorial := by
      field_simp; ring
    _ = 1 / (n - 2).factorial := by
      -- `(n-1)! = (n-2+1)! = (n-2)! * (n-1)`
      have : (n - 1 : ℕ) = (n - 2).succ := by omega
      rw [this, Nat.factorial_succ]; simp

/-- Step 5: Apply the simplification under the sum. -/
lemma apply_simplification :
  ∑' n : {n // 2 ≤ n}, n * ((n - 1 : ℝ) / n.factorial) =
    ∑' n : {n // 2 ≤ n}, 1 / (n.val - 2).factorial := by
  congr 1
  funext ⟨n, hn⟩
  exact simplify_factorial_expression n hn

/-- Step 6: Reindex the sum by `k = n - 2`. -/
lemma reindex_to_standard_form :
  ∑' n : {n // 2 ≤ n}, 1 / (n.val - 2).factorial = ∑' k : ℕ, 1 / k.factorial := by
  -- Use the general equivalence `Equiv.tsum_eq` on `shift`:
  -- shift : ℕ ≃ {n // 2 ≤ n},  k ↦ k+2
  let shift : ℕ ≃ {n // 2 ≤ n} where
    toFun    k := ⟨k + 2, Nat.le_add_right _ _⟩
    invFun   n := n.val - 2
    left_inv k := by simp [Nat.sub_add_cancel (Nat.le_add_right _ _)]
    right_inv n := by
      ext; simp [Nat.add_sub_cancel' n.prop]
  calc
    ∑' n : {n // 2 ≤ n}, 1/(n.val - 2).factorial
        = ∑' k : ℕ, 1/(shift.toFun k).val - 2.factorial := by
      -- Rewriting indices via `Equiv.tsum_eq`
      have hf : Summable fun n => (1 : ℝ) / (n.val - 2).factorial := by
        -- follows from summability of ∑1/k!
        simpa [f := fun k => _] using Summable.of_equiv (summable_inv_factorial : _)
      exact (Equiv.tsum_eq shift.symm hf).symm
    _ = ∑' k : ℕ, 1 / k.factorial := by
      funext; rfl

/-- Step 7: Recognize the exponential series. -/
lemma factorial_series_is_exp :
  ∑' k : ℕ, 1 / k.factorial = exp 1 := by
  exact exp_series_div_factorial 1

/-- Step 8: Put it all together in `calc`. -/
theorem main_result_complete : expected_hitting_time = exp 1 := by
  calc expected_hitting_time
      = ∑' n, n * hitting_time_pmf n := by exact expected_value_as_series
    _ = ∑' n, if n ≤ 1 then 0 else n * ((n - 1 : ℝ) / n.factorial) := by
      exact expand_pmf_in_sum
    _ = ∑' n : {n // 2 ≤ n}, n * ((n - 1 : ℝ) / n.factorial) := by
      exact (eliminate_zero_terms).symm
    _ = ∑' n : {n // 2 ≤ n}, 1 / (n.val - 2).factorial := by
      exact apply_simplification
    _ = ∑' k : ℕ, 1 / k.factorial := by
      exact reindex_to_standard_form
    _ = exp 1 := by
      exact factorial_series_is_exp
```

**Explanation of the Chain.**

1. **Unfold definition** of `expected_hitting_time`.
2. **Substitute** the PMF formula for $P(\tau=n)$.
3. **Eliminate** the terms $n=0,1$ which contribute 0.
4. **Simplify** the algebraic expression $n\,(n-1)/n! = 1/(n-2)!$.
5. **Reindex** the sum via the bijection $k\mapsto n-2$.
6. **Recognize** the classical series $\sum 1/k! = e$.

Each step invokes a supporting lemma, culminating in a single `calc` block that mirrors the informal derivation:

$$
E[\tau]
= \sum_{n}n\,P(\tau=n)
= \sum_{n\ge2}\frac{n(n-1)}{n!}
= \sum_{n\ge2}\frac1{(n-2)!}
= \sum_{k\ge0}\frac1{k!}
= e.
$$
